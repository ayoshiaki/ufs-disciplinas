\section{Busca BinĂˇria}

\begin{frame}{MotivaĂ§ĂŁo: Procurar em uma lista telefĂ´nica}
\begin{itemize}
\item Imagine procurar ``Kashiwabara'' em uma lista impressa com 1 milhĂŁo de nomes.
\item \textbf{Folha por folha?} Demoraria horas. Isso Ă© \alert{busca linear}: $\Theta(n)$.
\item \textbf{Abrir no meio e decidir lado?} Em $\sim 20$ comparaĂ§Ăµes achamos. Isso Ă© \alert{busca binĂˇria}: $\Theta(\lg n)$.
\end{itemize}
\begin{block}{PrĂ©-requisito crucial}
A busca binĂˇria sĂł funciona em arrays \textbf{ordenados}.
\end{block}
\end{frame}

\begin{frame}{ComparaĂ§ĂŁo numĂ©rica do ganho}
\centering
\begin{tabular}{rrr}
\toprule
$n$ & Linear ($n$) & BinĂˇria ($\lceil\lg n\rceil$) \\
\midrule
100 & 100 & 7 \\
1.000 & 1.000 & 10 \\
1.000.000 & 1.000.000 & 20 \\
1.000.000.000 & 1 bilhĂŁo & 30 \\
\bottomrule
\end{tabular}

\vspace{0.4cm}
\textbf{ReflexĂŁo:} dobrar $n$ adiciona apenas \emph{uma} comparaĂ§ĂŁo na busca binĂˇria.
\end{frame}

\begin{frame}[fragile]{Busca BinĂˇria: PseudocĂłdigo}
\begin{codebox}
\Procname{$\proc{Binary-Search}(A, v)$}
\li $\id{low} \gets 1$
\li $\id{high} \gets \attrib{A}{length}$
\li \While $\id{low} \leq \id{high}$
\li \Do $\id{mid} \gets \lfloor (\id{low}+\id{high})/2 \rfloor$
\li      \If $A[\id{mid}] \isequal v$
\li          \Then \Return $\id{mid}$
             \End
\li      \If $A[\id{mid}] < v$
\li          \Then $\id{low} \gets \id{mid}+1$
\li          \Else $\id{high} \gets \id{mid}-1$
             \End
     \End
\li \Return $\const{nil}$
\end{codebox}
\end{frame}

\begin{frame}{TraĂ§o passo a passo}
$A=[2,5,8,12,16,23,38,56,72,91]$,\ $v=23$
\vspace{0.3cm}

\begin{tabular}{ccccc}
\toprule
Iter & low & high & mid & $A[mid]$ vs 23 \\
\midrule
1 & 1 & 10 & 5 & 16 $<$ 23 $\Rightarrow$ low=6 \\
2 & 6 & 10 & 8 & 56 $>$ 23 $\Rightarrow$ high=7 \\
3 & 6 & 7 & 6 & 23 = 23 $\Rightarrow$ \textbf{retorna 6} \\
\bottomrule
\end{tabular}
\end{frame}

\begin{frame}{AnĂˇlise: por que $\Theta(\lg n)$?}
A cada iteraĂ§ĂŁo o intervalo cai pela metade: $n \to n/2 \to n/4 \to \cdots \to 1$.
ApĂłs $k$ iteraĂ§Ăµes: $n/2^k \leq 1 \Rightarrow k \geq \lg n$.

\begin{block}{Invariante}
Se $v \in A$, entĂŁo $v \in A[\id{low}..\id{high}]$.
\end{block}
\end{frame}

\begin{frame}{Prova de Corretude: Busca BinĂˇria}
\textbf{Invariante de loop (while):} Se $v$ estĂˇ em $A$, entĂŁo $v \in A[\id{low}..\id{high}]$.

\vspace{0.3cm}
\textbf{InicializaĂ§ĂŁo:} Antes da primeira iteraĂ§ĂŁo, $\id{low}=1$ e $\id{high}=n$. Se $v \in A$, claramente $v \in A[1..n]$. $\checkmark$

\vspace{0.3cm}
\textbf{ManutenĂ§ĂŁo:} Suponha o invariante verdadeiro no inĂ­cio de uma iteraĂ§ĂŁo.
\begin{itemize}
\item Se $A[\id{mid}]=v$, retornamos $\id{mid}$ (correto).
\item Se $A[\id{mid}]<v$: como $A$ Ă© ordenado, $v \notin A[\id{low}..\id{mid}]$. Logo, se $v \in A$, entĂŁo $v \in A[\id{mid}+1..\id{high}]$. Atualizamos $\id{low} \gets \id{mid}+1$. $\checkmark$
\item Se $A[\id{mid}]>v$: analogamente, $v \in A[\id{low}..\id{mid}-1]$ e $\id{high} \gets \id{mid}-1$. $\checkmark$
\end{itemize}

\vspace{0.3cm}
\textbf{TerminaĂ§ĂŁo:} Quando $\id{low}>\id{high}$, o intervalo Ă© vazio. Pelo invariante, se $v$ estivesse em $A$, estaria nesse intervalo vazio --- contradiĂ§ĂŁo. Logo, $v \notin A$ e retornar $\const{nil}$ Ă© correto. $\checkmark$
\end{frame}

\begin{frame}[fragile]{Busca BinĂˇria Recursiva}
\begin{codebox}
\Procname{$\proc{Binary-Search-Recursive}(A, v, low, high)$}
\li \If $\id{low} > \id{high}$
\li     \Then \Return $\const{nil}$
    \End
\li $\id{mid} \gets \lfloor (\id{low}+\id{high})/2 \rfloor$
\li \If $A[\id{mid}] \isequal v$
\li     \Then \Return $\id{mid}$
    \End
\li \If $A[\id{mid}] < v$
\li     \Then \Return $\proc{Binary-Search-Recursive}(A, v, \id{mid}+1, \id{high})$
\li     \Else \Return $\proc{Binary-Search-Recursive}(A, v, \id{low}, \id{mid}-1)$
    \End
\end{codebox}

\vspace{0.3cm}
\textbf{Chamada inicial:} $\proc{Binary-Search-Recursive}(A, v, 1, \attrib{A}{length})$
\end{frame}

\begin{frame}{VersĂŁo Iterativa vs Recursiva}
\begin{itemize}
\item \textbf{Complexidade:} ambas $\Theta(\lg n)$ no tempo.
\item \textbf{EspaĂ§o adicional:} iterativa usa $O(1)$; recursiva usa $O(\lg n)$ na pilha de chamadas.
\item \textbf{Clareza:} a recursiva Ă© mais declarativa e prĂłxima da definiĂ§ĂŁo matemĂˇtica.
\item \textbf{PrĂˇtica:} em geral, prefere-se a iterativa para evitar overhead de chamadas recursivas.
\end{itemize}

\begin{block}{RelaĂ§ĂŁo de recorrĂŞncia}
$T(n) = T(n/2) + \Theta(1) \Rightarrow T(n) = \Theta(\lg n)$ (Teorema Mestre, caso 2).
\end{block}
\end{frame}

\section{Insertion Sort}

\begin{frame}{Analogia: cartas na mĂŁo}
\begin{itemize}
\item MĂŁo esquerda: cartas jĂˇ \textbf{ordenadas}.
\item MĂŁo direita: pega prĂłxima carta e insere no lugar certo.
\item ApĂłs pegar todas, a mĂŁo esquerda estĂˇ ordenada.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{PseudocĂłdigo}
\begin{codebox}
\Procname{$\proc{Insertion-Sort}(A)$}
\li \For $j \gets 2$ \To $\attrib{A}{length}$
\li \Do $\id{key} \gets A[j]$
\li     $i \gets j-1$
\li     \While $i>0$ e $A[i]>\id{key}$
\li     \Do $A[i+1] \gets A[i]$
\li         $i \gets i-1$
        \End
\li     $A[i+1] \gets \id{key}$
    \End
\end{codebox}
\end{frame}

\begin{frame}{TraĂ§o completo: $A=[5,2,4,6,1,3]$}
\small
\begin{tabular}{cl}
\toprule
$j$ & Estado apĂłs inserir $A[j]$ \\
\midrule
2 & $[\mathbf{2,5},4,6,1,3]$ -- key=2 desloca 5 \\
3 & $[\mathbf{2,4,5},6,1,3]$ -- key=4 desloca 5 \\
4 & $[\mathbf{2,4,5,6},1,3]$ -- key=6 fica \\
5 & $[\mathbf{1,2,4,5,6},3]$ -- key=1 desloca tudo \\
6 & $[\mathbf{1,2,3,4,5,6}]$ -- key=3 \\
\bottomrule
\end{tabular}
\end{frame}

\begin{frame}{Complexidade}
\textbf{Melhor caso} (jĂˇ ordenado): while nunca executa $\Rightarrow \Theta(n)$.

\textbf{Pior caso} (decrescente): $\sum_{j=2}^n (j-1) = \frac{n(n-1)}{2} = \Theta(n^2)$.

\textbf{Caso mĂ©dio}: $\Theta(n^2)$.

\textbf{EspaĂ§o adicional}: $O(1)$, in-place, \textbf{estĂˇvel}.
\end{frame}

\begin{frame}{Prova de Corretude: Insertion Sort}
\textbf{Invariante de loop (for externo):} No inĂ­cio da iteraĂ§ĂŁo $j$, o subarray $A[1..j-1]$ contĂ©m os elementos originais de $A[1..j-1]$, mas em ordem crescente.

\vspace{0.25cm}
\textbf{InicializaĂ§ĂŁo:} Quando $j=2$, o subarray $A[1..1]$ tem um Ăşnico elemento, logo estĂˇ trivialmente ordenado. $\checkmark$

\vspace{0.3cm}
\textbf{ManutenĂ§ĂŁo:} Suponha $A[1..j-1]$ ordenado. Pegamos $\id{key}=A[j]$ e o inserimos na posiĂ§ĂŁo correta em $A[1..j-1]$:
\begin{itemize}
\item O loop while desloca elementos maiores que $\id{key}$ para a direita.
\item Ao final, inserimos $\id{key}$ na posiĂ§ĂŁo correta.
\item Portanto, $A[1..j]$ fica ordenado. $\checkmark$
\end{itemize}

\vspace{0.3cm}
\textbf{TerminaĂ§ĂŁo:} Quando $j=n+1$, o invariante garante que $A[1..n]$ estĂˇ ordenado. Como $n=\attrib{A}{length}$, o array completo estĂˇ ordenado. $\checkmark$
\end{frame}\begin{frame}{ExercĂ­cios -- Bloco 2}
\exbox{2.1}{Trace Insertion Sort em $[8,3,1,7,4,2,6,5]$ mostrando o array apĂłs cada $j$.}
\exbox{2.2}{Conte exatamente quantas comparaĂ§Ăµes e quantos deslocamentos ocorrem em $[5,4,3,2,1]$.}
\exbox{2.3}{Modifique para ordenar em ordem \emph{decrescente}.}
\exbox{2.4}{Por que Insertion Sort Ă© dito ``online''? DĂŞ um cenĂˇrio real onde isso Ă© Ăştil.}
\end{frame}

\section{Selection Sort}

\begin{frame}{Ideia}
A cada passo: encontre o \textbf{mĂ­nimo} do trecho nĂŁo ordenado e troque com a primeira posiĂ§ĂŁo nĂŁo ordenada.
\end{frame}

\begin{frame}[fragile]{PseudocĂłdigo}
\begin{codebox}
\Procname{$\proc{Selection-Sort}(A)$}
\li \For $i \gets 1$ \To $n-1$
\li \Do $\id{min} \gets i$
\li     \For $j \gets i+1$ \To $n$
\li     \Do \If $A[j]<A[\id{min}]$
\li             \Then $\id{min} \gets j$
            \End
        \End
\li     trocar $A[i] \leftrightarrow A[\id{min}]$
    \End
\end{codebox}
\end{frame}

\begin{frame}{TraĂ§o: $A=[5,2,4,6,1,3]$}
\small
$i=1$: min=1 $\Rightarrow [\mathbf{1},2,4,6,5,3]$ \\
$i=2$: min=2 $\Rightarrow [1,\mathbf{2},4,6,5,3]$ \\
$i=3$: min=6 $\Rightarrow [1,2,\mathbf{3},6,5,4]$ \\
$i=4$: min=6 $\Rightarrow [1,2,3,\mathbf{4},5,6]$ \\
$i=5$: min=5 $\Rightarrow [1,2,3,4,\mathbf{5},6]$
\end{frame}

\begin{frame}{Complexidade e caracterĂ­sticas}
Sempre $\Theta(n^2)$ comparaĂ§Ăµes (independe da entrada).
Apenas $O(n)$ trocas -- Ăştil quando \emph{escrever} Ă© caro (memĂłria flash).
\textbf{NĂŁo Ă© estĂˇvel}. In-place.
\end{frame}

\begin{frame}[shrink]{Prova de Corretude: Selection Sort}
\textbf{Invariante de loop (for externo):} No inĂ­cio da iteraĂ§ĂŁo $i$, o subarray $A[1..i-1]$ contĂ©m os $(i-1)$ menores elementos de $A$ em ordem crescente, e $A[i..n]$ contĂ©m os elementos restantes.

\vspace{0.3cm}
\textbf{InicializaĂ§ĂŁo:} Quando $i=1$, o subarray $A[1..0]$ Ă© vazio, logo o invariante Ă© trivialmente verdadeiro. $\checkmark$

\vspace{0.3cm}
\textbf{ManutenĂ§ĂŁo:} Suponha o invariante vĂˇlido para $i$. O loop interno encontra o Ă­ndice $\id{min}$ do menor elemento em $A[i..n]$. Ao trocar $A[i] \leftrightarrow A[\id{min}]$:
\begin{itemize}
\item $A[1..i]$ contĂ©m os $i$ menores elementos em ordem crescente.
\item $A[i+1..n]$ contĂ©m os elementos restantes. $\checkmark$
\end{itemize}

\vspace{0.3cm}
\textbf{TerminaĂ§ĂŁo:} Quando $i=n$, temos $A[1..n-1]$ com os $(n-1)$ menores elementos ordenados. O Ăşltimo elemento $A[n]$ Ă© automaticamente o maior. Logo, $A[1..n]$ estĂˇ ordenado. $\checkmark$
\end{frame}

\end{frame}

\section{Bubble Sort}

\begin{frame}[fragile]{Ideia e pseudocĂłdigo}
Compare pares adjacentes; troque se fora de ordem. Repita.
\begin{codebox}
\Procname{$\proc{Bubble-Sort}(A)$}
\li \For $i \gets 1$ \To $n-1$
\li \Do \For $j \gets 1$ \To $n-i$
\li     \Do \If $A[j]>A[j+1]$
\li             \Then trocar $A[j] \leftrightarrow A[j+1]$
            \End
        \End
    \End
\end{codebox}
\end{frame}

\begin{frame}{TraĂ§o: $A=[5,1,4,2,8]$, passagem 1}
$[5,1,4,2,8] \to [1,5,4,2,8] \to [1,4,5,2,8] \to [1,4,2,5,8] \to [1,4,2,5,8]$

O 8 jĂˇ estava no lugar; o 5 ``borbulhou'' atĂ© a posiĂ§ĂŁo correta entre os nĂŁo-ordenados.
\end{frame}

\begin{frame}{OtimizaĂ§ĂŁo: flag de troca}
Se uma passagem inteira nĂŁo fizer trocas $\Rightarrow$ array jĂˇ ordenado, pare.
Melhor caso passa de $\Theta(n^2)$ para $\Theta(n)$.
\end{frame}

\begin{frame}[shrink]{Prova de Corretude: Bubble Sort}
\textbf{Invariante de loop (for externo):} No inĂ­cio da iteraĂ§ĂŁo $i$, os $i-1$ maiores elementos estĂŁo nas posiĂ§Ăµes $A[n-i+2..n]$ em ordem crescente.

\vspace{0.3cm}
\textbf{InicializaĂ§ĂŁo:} Quando $i=1$, o subarray $A[n+1..n]$ Ă© vazio, logo o invariante Ă© trivialmente verdadeiro. $\checkmark$

\vspace{0.3cm}
\textbf{ManutenĂ§ĂŁo:} Suponha o invariante vĂˇlido para $i$. O loop interno percorre $A[1..n-i]$ comparando pares adjacentes e trocando se necessĂˇrio:
\begin{itemize}
\item O maior elemento de $A[1..n-i+1]$ ``borbulha'' atĂ© a posiĂ§ĂŁo $n-i+1$.
\item Combinado com o invariante, temos os $i$ maiores elementos em $A[n-i+1..n]$ ordenados. $\checkmark$
\end{itemize}

\vspace{0.3cm}
\textbf{TerminaĂ§ĂŁo:} Quando $i=n$, os $(n-1)$ maiores elementos estĂŁo em $A[2..n]$ ordenados. O primeiro elemento $A[1]$ Ă© automaticamente o menor. Logo, $A[1..n]$ estĂˇ ordenado. $\checkmark$
\end{frame}
